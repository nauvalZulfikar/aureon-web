# Aureon ML Pipeline v2.4
 
from aureon import Pipeline
from aureon.ml import AutoML
 
# Init pipeline
pipe = Pipeline(env="prod")
 
# Connect sources
pipe.ingest([
  "postgres://db:5432",
  "s3://data-lake/raw"
])
 
# Train & deploy
model = AutoML.fit(pipe)
model.deploy(target="api")
